CREATE PROCEDURE [dbo].[InsertEmployeeProcdure]
       @empno int, 
       @empname varchar(200), 
       @empsal numeric(10,2),
	   @emptype varchar(200)
AS
IF NOT EXISTS (SELECT NULL FROM Code_Employee
                WHERE @empno =@empno)
BEGIN
    INSERT INTO 
        Code_Employee 
        (empno,empname,empsal,emptype)

END
